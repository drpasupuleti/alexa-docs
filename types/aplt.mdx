Types for devices with character displays (e.g., Echo Dot with clock). APLT documents render content on LED 7-segment alphanumeric displays.

## APLTDocument

```typescript
interface APLTDocument {
  type: "APLT";
  version: string;
  description?: string;
  mainTemplate: APLTMainTemplate;
}

interface APLTMainTemplate {
  parameters?: string[];
  item?: APLTComponent;
  items?: APLTComponent[];
}
```

## APLTComponent

```typescript
interface APLTComponent {
  type: APLTComponentType;
  id?: string;
  when?: boolean | string;
  description?: string;
  text?: string;                                    // Text
  textAlign?: "left" | "right" | "center";          // Text
  overflow?: "clip" | "marquee";                    // Text
  items?: APLTComponent[];                          // Container, Pager
  item?: APLTComponent;                             // Container, Pager
  format?: string;                                  // TimeText
}

type APLTComponentType = "Text" | "Container" | "Pager" | "TimeText";
```

## APLTCommand

```typescript
interface APLTCommand {

  delay?: number;
  when?: boolean | string;
  componentId?: string;

```

## Directives

interface APLTRenderDocumentDirective extends Directive {
  type: "Alexa.Presentation.APLT.RenderDocument";

interface APLTExecuteCommandsDirective extends Directive {
