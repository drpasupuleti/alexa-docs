Types for devices with screens (Echo Show, Fire TV). Covers documents, components, commands, data sources, five directive types, and four request types.

## APLDocument

```typescript
interface APLDocument {
  type: "APL" | "Link";
  version?: string;
  src?: string;                              // Link-type only
  description?: string;
  theme?: string;
  import?: APLImport[];
  resources?: APLResource[];
  styles?: Record<string, unknown>;
  layouts?: Record<string, unknown>;
  mainTemplate?: APLMainTemplate;
  onMount?: APLCommand[];
  settings?: Record<string, unknown>;
  extensions?: APLExtensionRequest[];
}

interface APLImport { name: string; version: string; source?: string; }
interface APLMainTemplate { parameters?: string[]; items?: APLComponent[]; item?: APLComponent; }
interface APLExtensionRequest { name: string; uri: string; }
```

## APLComponent

```typescript
interface APLComponent {
  type: string;
  id?: string;
  when?: boolean | string;
  display?: "normal" | "invisible" | "none";
  height?: string | number;
  width?: string | number;
  opacity?: number;
  style?: string;
  bind?: APLBind[];
  items?: APLComponent[];
  // ... many more properties
}

interface APLBind { name: string; value: unknown; type?: string; }
```

## APLCommand

```typescript
interface APLCommand {
  type: string;
  description?: string;
  delay?: number;
  when?: boolean | string;
  screenLock?: boolean;
  sequencer?: string;
}
```

## Data Sources

```typescript
type APLDataSource = ObjectDataSource | DynamicIndexListDataSource | DynamicTokenListDataSource;

interface ObjectDataSource {
  type: "object";
  objectId?: string;
  properties?: Record<string, unknown>;
  transformers?: DataSourceTransformer[];
}

interface DynamicIndexListDataSource {
  type: "dynamicIndexList";
  listId: string;
  startIndex: number;
  minimumInclusiveIndex?: number;
  maximumExclusiveIndex?: number;
  items: Record<string, unknown>[];
}

interface DynamicTokenListDataSource {
  type: "dynamicTokenList";
  listId: string;
  pageToken: string;
  items: Record<string, unknown>[];
  forwardPageToken?: string;
  backwardPageToken?: string;
}
```
