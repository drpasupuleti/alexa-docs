## Quick Example

```typescript
import { validateDirective } from "alexa-apl-interface";
import { isRequestFresh } from "alexa-apl-interface";
import { isValidDocumentName } from "alexa-apl-interface";

// Validate a directive before sending
const result = validateDirective(myDirective);
if (!result.valid) {
  console.error("Directive errors:", result.errors);
}

// Check request freshness
if (!isRequestFresh(alexaRequest)) {
  return createAPLResponse({ speech: "Request expired. Please try again." });
}

// Validate a document name
if (!isValidDocumentName(userProvidedName)) {
  throw new Error("Invalid document name");
}
```
